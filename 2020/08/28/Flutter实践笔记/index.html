导航栏自定义

  • 必须是一个statefuleWidget

  • 实现抽象类PreferredSizeWidget，并实现抽象方法Size get preferredSize

  • 返回的是一个PreferredSize Widget

实例

/// 以下是核心代码，UI构建的代码省略

class MyNavigationBar extends StatefulWidget implements PreferredSizeWidget {
  @override
  _MyNavigationBarState createState() => _MyNavigationBarState();

  Size get preferredSize {
    /// 优先尺寸
    /// 尺寸返回的高度是指导航栏的高度，不包括状态栏的高度，所以正常的都是返回44
    /// 备注：double.infinity表示尽可能的大，所以这里等于全屏的宽度
    return Size(double.infinity, 44);
  }
}


class _RWStudyNavigationBarState extends State<RWStudyNavigationBar> {
  @override
  void initState() {
    super.initState();
  }

  @override
  Widget build(BuildContext context) {
    return PreferredSize(
      /// 这里设置的尺寸同上，不过这里既是设置为0，也没关系，实际以上面设置的尺寸为优先
      /// 由于这里是必传参数所以就按照上面的写，统一比较好
      preferredSize: Size(double.infinity, 44),
      child: Container(
        /// 这里推荐在子widge第一层，指定高度，并且指定的高度与设置的优先尺寸一致
        /// 这里可以直接设置高度为:double.infinity，高度尽快能搞，那么其值就等于前面设置的preferredSize的高度
        height: double.infinity,
        color: Theme.of(context).primaryColor,
        child: Row(
          crossAxisAlignment: CrossAxisAlignment.end,
          /// row的主轴对齐方式选择：MainAxisAlignment.spaceBetween
          /// 可以让子widget左右两边就紧靠着边缘对齐
          /// 再配合Expanded(),就能准确的布局子widget的位置
          mainAxisAlignment: MainAxisAlignment.spaceBetween,
          children: <Widget>[
            /// 具体子Widet代码
          ],
        ),
      ),
    );
  }
}

TabBar、TabBarView、TabController 构建分段视图

比如今日头条的分段视图

实例

/// 以下是核心代码，忽略其细节

class StudyPage extends StatefulWidget {
  @override
  _StudyPageState createState() => _StudyPageState();
}

class _StudyPageState extends State<StudyPage> with SingleTickerProviderStateMixin {
  /// 1.0 定义TabController，这是将TabBar与TabBarView关联的关键
  TabController _controller;
  List<String> _titles;

  @override
  void initState() {
    super.initState();
    _titles = ['关注', '首页', '学科'];
    /// 2.0 初始化TabController
    /// initialIndex：默认选择下标
    /// length：个数
    /// vsync：固定写法(一定要在StatefulWidget中)
    _controller = TabController(initialIndex: 1, length: _titles.length, vsync: this);
  }

  @override
  void dispose() {
    super.dispose();
    /// 最后销毁TabController
    _controller.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
        appBar: RWStudyNavigationBar(controller: _controller, titles: _titles),
      	/// 3.0 TabBarView定义分段视图的页面
	      /// 并将TabController传入
      	/// 页面的数量必须与TabController定义的长度length一致，否则报错
        body: TabBarView(
          controller: _controller,
          children: <Widget>[
            RWStudyFoucsPage(),
            RWStudyHomePage(),
            RWStudySubjectPage()
          ],
        ));
  }
}


/// RWStudyNavigationBar 核心代码，其余无关代码都删除，避免混淆
class RWStudyNavigationBar extends StatefulWidget implements PreferredSizeWidget {
  final TabController controller;
  final List<String> titles;
  RWStudyNavigationBar({@required this.controller, @required this.titles});

  @override
  _RWStudyNavigationBarState createState() => _RWStudyNavigationBarState();

  Size get preferredSize {
    return Size(RWScreenFit.screenWidth, 44);
  }
}

class _RWStudyNavigationBarState extends State<RWStudyNavigationBar> {
  @override
  Widget build(BuildContext context) {
    return PreferredSize(
      preferredSize: Size(double.infinity, 44),
      child: Container(
        height: double.infinity,
        color: Theme.of(context).primaryColor,
        child: Row(
          crossAxisAlignment: CrossAxisAlignment.end,
          mainAxisAlignment: MainAxisAlignment.spaceBetween,
          children: <Widget>[
            Expanded(
              child: Container(
                  padding: EdgeInsets.only(left: 10.px),
									/// 4.0 定义TabBar样式
                	/// 并将TabController传入
                	child: TabBar(
                    indicatorSize: TabBarIndicatorSize.label,	/// 指示器尺寸与文字相同
                    indicatorColor: Colors.white,	/// 指示器颜色
                    indicatorWeight: 3.0.px,	/// 指示器宽度
                    labelColor: Colors.white,	/// 选中：文字颜色
                    labelStyle: TextStyle(fontSize: 16.px),	/// 选中：文字样色
                    labelPadding: EdgeInsets.fromLTRB(10.px, 0.0, 10.px, 0.0), /// 文字内边距
                    unselectedLabelColor: Colors.white38,	/// 未选中：文字颜色
                    unselectedLabelStyle: TextStyle(fontSize: 16.px),	/// 未选中：文字样色
                    controller: widget.controller,	/// 传入TabController
                    isScrollable: true,
                    /// 5.0 使用Tab定义分段标题
                    tabs: widget.titles.map((e) => Tab(text: e)).toList(),
                  )),
            ),
          ],
        ),
      ),
    );
  }
}

/// easyRefresh初始化有三种方式
/// 第一种
EasyRefresh(child:)
/// 第二种
EasyRefresh.custom(slivers:)
/// 第三种
EasyRefresh.builder(
    builder: (context, physics, header, footer) {
      return CustomScrollView(
        physics: physics,
        slivers: <Widget>[],
      );
    }
}
  
/// 第二种和第三种要求传入的widget都是Sliver类型的
/// 如果不是Sliver类型，建议使用第一种方式
  
/// 完成下拉刷新上拉加载的功能，以下是核心代码，以第一种方式为例子
EasyRefresh(
 			/// 必须要传入一个EasyRefreshController对象，关键
  		/// 这是控制刷新和加载的状态
      controller: _controller,
 			/// 是否第一次刷新
      firstRefresh: true,
  		/// 刷新和加载状态独立（推荐设置为true，否则如果加载或刷新在进行中，那么另外一个则会失效）
      taskIndependence: true,
  		/// 是否开启控制结束刷新 (这是必须设置true，这样controller才能控制刷新状态)
      enableControlFinishLoad: true,
  		/// enableControlFinishLoad (这是必须设置true，这样controller才能控制加载状态)
      enableControlFinishRefresh: true,
      child: ListView(
        children: [
        ],
      ),
		  /// 下拉刷新触发方法
      onRefresh: () async {
        await _viewModel.requestRecommentArticleList((data) {
          setState(() {
            articleList.clear();
            articleList.addAll(data);
            /// 更新刷新的状态
            _controller.finishRefresh(success: true);
          });
        }, (code, msg) {
          _controller.finishLoad(success: false);
        });
      },
  		/// 上拉加载触发方法
      onLoad: () async {
        await _viewModel.requestNextPageRecommentArticleList((data) {
          setState(() {
            articleList.addAll(data);
            /// 更新加载的状态
            _controller.finishLoad(success: true);
            /// 判断是否还有更多加载
            _controller.finishLoad(noMore: (data as List).length < 20);
          });
        }, (code, msg) {
          _controller.finishLoad(success: false);
        });
      },
    )

/// ListView 嵌套 ListView 会出现一个报错'HasSize'
/// 必须将嵌套的子ListView，设置以下属性
/// 设置为true代表ListView会根据指示图的尺寸而定，子widget的尺寸是多少就占据多少
/// shrinkWrap: true

class RWStudySubjectPage extends StatefulWidget {
  RWStudySubjectPage({Key key}) : super(key: key);

  @override
  _RWStudySubjectPageState createState() => _RWStudySubjectPageState();
}

class _RWStudySubjectPageState extends State<RWStudySubjectPage> {
  @override
  Widget build(BuildContext context) {
    return ListView(
      children: [
        Container(
          height: 50,
          color: Colors.red,
        ),
        ListView.builder(
          /// 不尽可能扩大
          shrinkWrap: true,
          /// 取消滑动交互
          physics: NeverScrollableScrollPhysics(),
          itemCount: 20,
          itemBuilder: (context, index) {
            return ListTile(
              title: Text('这是第 $index 行'),
            );
          },
        )
      ],
    );
  }
}

时间今天、昨天、前天的判断

/// 今天、昨天、前天
static timeFormatDaysAgo(int timestamp) {
  DateTime targetday = DateTime.fromMillisecondsSinceEpoch(timestamp);

  /// 今天
  DateTime today = DateTime.now();

  /// 今天的23：59：59
  DateTime todayLast =
    DateTime(today.year, today.month, today.day, 23, 59, 59);

  /// 时间对比
  Duration diff = todayLast.difference(targetday);

  if (diff < Duration(days: 1)) {
    return '今天';
  } else if (diff >= Duration(days: 1) && diff < Duration(days: 2)) {
    return '昨天';
  } else if (diff >= Duration(days: 2) && diff < Duration(days: 3)) {
    return '前天';
  } else {
    return timeFormatForYYMMDD(timestamp);
  }
}
